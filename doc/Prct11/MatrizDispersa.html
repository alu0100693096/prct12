def []=(fila, columna, valor)
  if valor != zero
    @container[Posicion.new(fila, columna)] = valor
  else
    @container.delete(Posicion.new(fila, columna))
  end
end

def columnas=(value)
  validate_sizes(value)

  if value < columnas
    @container.each_key do |k|
      if k.c >= value
        @container.delete k
      end
    end
  end

  columnas = value
end

def filas=(value)
  validate_sizes(value)

  if value < filas
    @container.each_key do |k|
      if k.f >= value
        @container.delete k
      end
    end
  end

  filas = value
end
