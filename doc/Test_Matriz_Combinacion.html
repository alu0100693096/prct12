def test_enteros
    #Suma
    @m3[0, 0] = 2
    @m3[0, 1] = 4
    @m3[1, 0] = 4
    @m3[1, 1] = 6
    assert_equal(@m3, @m1+@m2)

    #Resta
    @m3[0, 0] = 2
    @m3[0, 1] = 4
    @m3[1, 0] = -2
    @m3[1, 1] = 6
    assert_equal(@m3, @m1-@m2)

    #Multiplicaci√≥n
    @m3[0, 0] = 12
    @m3[0, 1] = 0
    @m3[1, 0] = 18
    @m3[1, 1] = 0
    assert_equal(@m3, @m1*@m2)

    #Comparaci√≥n
    @m3[0, 0] = 0
    @m3[0, 1] = 0
    @m3[1, 0] = 3
    @m3[1, 1] = 0
    assert_equal(@m3, @m2)
end
